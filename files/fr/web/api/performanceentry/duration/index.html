{{APIRef("Performance Timeline API")}}

La propriété duration renvoie un {{domxref("DOMHighResTimeStamp", "timestamp")}} qui correspond à la durée de l'{{domxref("PerformanceEntry", "entrée de performance","",1)}}.

{{AvailableInWorkers}}

La valeur retournée par cette propriété dépend du {{domxref("PerformanceEntry.entryType", "type")}} de l'entrée de performance :

  • « frame » - retourne un {{domxref("DOMHighResTimeStamp", "timestamp")}} indiquant la différence entre le startTime de deux vues successives.
  • « mark » - retourne "0" (un marquage n'a pas de durée).
  • « measure » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} qui correspond à la durée de la mesure.
  • « navigation » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} qui est respectivement la différence entre les propriétés {{domxref("PerformanceNavigationTiming.loadEventEnd")}} et {{domxref("PerformanceEntry.startTime")}}.
  • « resource » - renvoie la différence entre {{domxref("PerformanceEntry.responseEnd", "responseEnd")}} {{domxref("DOMHighResTimeStamp", "timestamp")}} (le moment de la fin de la réponse) de la ressource et son {{domxref("PerformanceEntry.startTime", "startTime")}} {{domxref("DOMHighResTimeStamp", "timestamp")}} (le point de départ).

Cette propriété est en {{readonlyInline}}.

Syntaxe

entry.duration;

Valeur de retour

Un objet {{domxref("DOMHighResTimeStamp")}} représentant la durée de l'{{domxref("PerformanceEntry","entrée de performance","",1)}}. Si le concept de durée ne s'applique pas à une mesure de performance particulière, le navigateur peut choisir de renvoyer une durée de 0.

Note : si l'entrée de performance a un {{domxref("PerformanceEntry.entryType", "entryType")}} "resource" (c-à-d. l'entrée est un objet {{domxref("PerformanceResourceTiming")}}), cette propriété renvoie la différence entre les {{domxref("PerformanceEntry.responseEnd")}} et {{domxref("PerformanceEntry.startTime")}} {{domxref("DOMHighResTimeStamp", "timestamps")}}.

Exemple

L'exemple suivant montre l'utilisation de la propriété duration.

function run_PerformanceEntry() {
  console.log("Support de PerformanceEntry ...");

  if (performance.mark === undefined) {
    console.log("... performance.mark N'est pas pris en charge");
    return;
  }

  // Créer quelques entrées de performance via la méthode mark()
  performance.mark("Begin");
  do_work(50000);
  performance.mark("End");

  // Utilise getEntries() pour itérer à travers chaque entrée.
  let p = performance.getEntries();
  for (let i=0; i < p.length; i++) {
    console.log("Entry[" + i + "]");
    check_PerformanceEntry(p[i]);
  }
}
function check_PerformanceEntry(obj) {
  let properties = ["name", "entryType", "startTime", "duration"];
  let methods = ["toJSON"];

  for (let i=0; i < properties.length; i++) {
    // check each property
    let supported = properties[i] in obj;
    if (supported)
      console.log("..." + properties[i] + " = " + obj[properties[i]]);
    else
      console.log("..." + properties[i] + " = N'est pas pris en charge");
  }
  for (let i=0; i < methods.length; i++) {
    // check each method
    let supported = typeof obj[methods[i]] == "function";
    if (supported) {
      let js = obj[methods[i]]();
      console.log("..." + methods[i] + "() = " + JSON.stringify(js));
    } else {
      console.log("..." + methods[i] + " = N'est pas pris en charge");
    }
  }
}
