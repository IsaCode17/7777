{{APIRef("Performance Timeline API")}}

La propriété startTime renvoie le premier {{domxref("DOMHighResTimeStamp", "timestamp")}} enregistré de l{{domxref("PerformanceEntry", "entrée de performance")}}.

{{AvailableInWorkers}}

La valeur renvoyée par cette propriété dépend du {{domxref("PerformanceEntry.entryType", "type")}} de l'entrée de performance :

  • « frame » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} quand l'affichage a été démarré.
  • « mark » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} lorsque le marquage a été créé par un appel de {{domxref("Performance.mark", "performance.mark()")}}.
  • « measure » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} lorsque la mesure a été créée par un appel de {{domxref("Performance.measure", "performance.measure()")}}.
  • « navigation » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} avec une valeur de "0".
  • « resource » - retourne le {{domxref("DOMHighResTimeStamp", "timestamp")}} immédiatement avant que le navigateur ne {{domxref("PerformanceEntry.fetchStart","commence à récupérer la ressource","",1)}}.

Cette propriété est en {{readonlyInline}}.

Syntaxe

entry.startTime;

Valeur de retour

Un objet {{domxref("DOMHighResTimeStamp")}} représentant le premier horodatage lorsque l'{{domxref("PerformanceEntry", "entrée de performance")}} a été créée.

Note : Si l'entrée de performance a un {{domxref("PerformanceEntry.entryType", "entryType")}} "resource" (c-à-d. que l'entrée est un objet {{domxref("PerformanceResourceTiming")}}), cette propriété renvoie le {{domxref("PerformanceEntry.fetchStart")}} {{domxref("DOMHighResTimeStamp", "timestamp")}}.

Exemple

L'exemple suivant montre l'utilisation de la propriété startTime.

function run_PerformanceEntry() {
  console.log("Support de PerformanceEntry ...");

  if (performance.mark === undefined) {
    console.log("... performance.mark n'est pas pris en charge");
    return;
  }

  // Crée quelques entrées de performance via la méthode mark()
  performance.mark("Begin");
  do_work(50000);
  performance.mark("End");

  // Utilise getEntries() pour itérer à travers chaque entrée
  let p = performance.getEntries();
  for (let i=0; i < p.length; i++) {
    log("Entry[" + i + "]");
    check_PerformanceEntry(p[i]);
  }
}
function check_PerformanceEntry(obj) {
  let properties = ["name", "entryType", "startTime", "duration"];
  let methods = ["toJSON"];

  for (let i=0; i < properties.length; i++) {
    // check each property
    let supported = properties[i] in obj;
    if (supported)
      log("..." + properties[i] + " = " + obj[properties[i]]);
    else
      log("..." + properties[i] + " = N'est pas pris en charge");
  }
  for (let i=0; i < methods.length; i++) {
    // check each method
    let supported = typeof obj[methods[i]] == "function";
    if (supported) {
      let js = obj[methods[i]]();
      log("..." + methods[i] + "() = " + JSON.stringify(js));
    } else {
      log("..." + methods[i] + " = N'est pas pris en charge");
    }
  }
}
